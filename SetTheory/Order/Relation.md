# binary relation 'R on A'
A = {0, 1, 2}

## 集合A上の二項関係R
R = {(x,y) | (x,y) ∈ A × A, x < y}  -- ?  
  = {(0,1), (1,2)}

## 逆関係
R^(-1) = {(y,x) | (x,y) ∈ R}
       = {(1,0), (2,1)}

## 関係の合成
R ○ R' = {(x,b) | (x,y) ∈ R, (a,b) ∈ R', y = a}
- ex
    - R' = R^(-1) ⇒ R ○ R' = {(0,0), (1,1)}
